artY;
  if(Math.abs(dx)+Math.abs(dy)>3) moved=true;
  shell.scrollLeft=startLeft-dx;
  shell.scrollTop=startTop-dy;
  e.preventDefault();
}
function endPan(e){
  if(!dragging) return;
  dragging=false;
  shell.classList.remove('dragging');
  try{ if(pointerId!==null) shell.releasePointerCapture(pointerId); }catch(_e){}
  pointerId=null;
}
shell.addEventListener('pointerdown', beginPan, {passive:false});
shell.addEventListener('pointermove', movePan, {passive:false});
shell.addEventListener('pointerup', endPan);
shell.addEventListener('pointercancel', endPan);
shell.addEventListener('pointerleave', e=>{ if(dragging && e.pointerType==='mouse') endPan(e); });
shell.addEventListener('click', e=>{ if(moved){ e.preventDefault(); e.stopPropagation(); moved=false; } }, true);

// Roda do mouse: zoom ancorado no ponto do cursor. Shift + roda: pan horizontal. Alt + roda: pan vertical.
shell.addEventListener('wheel', e=>{
  e.preventDefault();
  if(e.shiftKey){ shell.scrollLeft += e.deltaY; return; }
  if(e.altKey){ shell.scrollTop += e.deltaY; return; }
  const before=z;
  const delta=e.deltaY<0 ? .08 : -.08;
  const rect=shell.getBoundingClientRect();
  const mx=e.clientX-rect.left+shell.scrollLeft;
  const my=e.clientY-rect.top+shell.scrollTop;
  setZoom(z+delta);
  const ratio=z/before;
  shell.scrollLeft=(mx*ratio)-(e.clientX-rect.left);
  shell.scrollTop=(my*ratio)-(e.clientY-rect.top);
}, {passive:false});
